# Dependencias
Instalar las siguientes dependecias:
- npm i express
- npm i cors
- npm i nodemon -D
- npm i mysql2
- npm i body-parser
- npm i dotenv -save

# .env
Cambiar a sus necesidades
- PORT = 4000
- HOST_DB = localhost
- PORT_DB = 3306
- USER_DB = root
- PASSWORD_DB = ALANHUEVOs18
- NAME_DB = resiliente
